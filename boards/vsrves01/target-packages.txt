# Packages cross-emerged into /target.  This is the single source of
# truth for what ends up on the board: override-kernel.sh runs
# `emerge -e @system` to rebuild the stock stage3 with our CFLAGS (no
# C extension), then `emerge` over this list, then `mksquashfs /target`
# directly — no separate cherry-pick list.
#
# Phram-rootfs architecture (see override-assemble.sh): the sqfs is
# loaded by vendor's DDRLoad to a 64 MiB DDR region at 0x84000000 and
# mounted by the kernel via phram+squashfs.  We have room for a full
# Gentoo userland — no need to lean on busybox applets like the cpio
# embed path did.

# Init chain.  See target-packages.use for the openrc USE-flag dance
# that keeps systemd-utils out of the resolve on rv32-musl.
sys-apps/baselayout
sys-apps/sysvinit
sys-apps/openrc

# busybox 1.38: kept for emergency rescue + occasional `busybox foo`
# applet use.  1.36.x fails to build on Linux 6.x rv32 (kernel dropped
# 32-bit-only time syscalls for Y2038; 1.36 hwclock.c / i2c_tools.c
# reference them unconditionally).  1.38 is unkeyworded — `**` accepts.
=sys-apps/busybox-1.38.0 **

# musl libc.
sys-libs/musl

# DHCP client = busybox udhcpc (NOT net-misc/dhcpcd — that's heavy +
# multi-second event loop spins the rv32 core at 100% before timing
# out on the AUI ethernet driver's link-up notification).  Driven by
# /etc/local.d/network.start (in initramfs-overlay) which calls
# `udhcpc -i eth0 -b -t 5 -T 2`.
